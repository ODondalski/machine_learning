def generate_random_iris():
    data_temp = pd.DataFrame(iris.data, columns = iris.feature_names)
    test = []
    for i in range(30):
        test.append([])
        random_value = random.uniform(data_temp['sepal length (cm)'].min(), data_temp['sepal length (cm)'].max())
        test[i].append(round(random_value, 1))
        random_value = random.uniform(data_temp['sepal width (cm)'].min(), data_temp['sepal width (cm)'].max())
        test[i].append(round(random_value, 1))
        random_value = random.uniform(data_temp['petal length (cm)'].min(), data_temp['petal length (cm)'].max())

        random_value = random.uniform(data_temp['petal width (cm)'].min(), data_temp['petal width (cm)'].max())
        test[i].append(round(random_value, 1))
    return test

class KnnClass:
    def __init__(self, data, group_test=0):
        self.data = data
        self.data['group'] = ''
        self.data.loc[:49, 'group'] = 'A'
        self.data.loc[50:99, 'group'] = 'B'
        self.data.loc[100:149, 'group'] = 'C'
        train, test = train_test_split(self.data, test_size=0.2)
        self.test_copy = test.copy()
        self.data = train.copy()
        train.drop('group', axis=1, inplace=True)
        self.x = train.values
        # number_of_elements = len(data)
        # number_of_groups = 2
        # self.k = self.define_k(number_of_elements, number_of_groups)
        self.k = 3
        self.accuracy = 0
        test.drop('group', axis=1, inplace=True)
        if group_test == 0:
            self.group_test = test.values.tolist()
        else:
            self.group_test = group_test
        self.test_data = pd.DataFrame(self.group_test, columns = iris.feature_names)
        self.test_data['group'] = ''
    
    def define_k(self, n, c):
        k = ceil(np.sqrt(n))
        while k <= c:
            k += 1
        if c % 2 == 0 and k % 2 == 0:
            k += 1
        elif c % 2 == 1 and k % 2 == 1:
            k += 1
        return k

    def euclides_distance_func(self):
        distance_list = []
        distance = np.zeros(len(self.x))
        for z in range(len(self.group_test)):
            for i in range(len(self.x)):
                for j in range(len(self.x[0])):
                    value = ((self.x[i][j] - self.group_test[z][j]) ** 2)
                    distance[i] += round(value, 2)
                distance[i] = round(np.sqrt(distance[i]), 2)
            distance_list.append(distance)
            distance = np.zeros(len(self.x))
        assert(len(distance_list) == len(self.group_test))
        return distance_list
    
    def cosine_distance_func(self):
        distance_list = []
        distance = np.zeros(len(self.x))
        for z in range(len(self.group_test)):
            for i in range(len(self.x)):
                dot_product = np.dot(self.x[i], self.group_test[z])
                norm_vector1 = np.linalg.norm(self.x[i])
                norm_vector2 = np.linalg.norm(self.group_test[z])
                cosine_distance = 1 - dot_product / (norm_vector1 * norm_vector2)
                distance[i] += cosine_distance
            distance_list.append(distance)
            distance = np.zeros(len(self.x))
        assert(len(distance_list) == len(self.group_test))
        return distance_list
    
    def manhattan_distance_func(self):
        distance_list = []
        distance = np.zeros(len(self.x))
        for z in range(len(self.group_test)):
            for i in range(len(self.x)):
                distance[i] += np.round(np.sum(np.abs(self.x[i] - self.group_test[z])), 3)
            distance_list.append(distance)
            distance = np.zeros(len(self.x))
        assert(len(distance_list) == len(self.group_test))
        return distance_list
    
    def knn_calculation(self, dist_func):
        cp_test_groups = self.test_copy['group'].values
        correct = 0
        dist = dist_func
        k = self.k
        for i in range(len(self.group_test)):
            self.data['distance_from_test'] = dist[i]
            data_temp = self.data.sort_values('distance_from_test', ascending=True)
            k_nearest = data_temp[:k]
            inverse_list = 1 / k_nearest['distance_from_test']
            groups = k_nearest['group']
            groups = groups.values.tolist()
            inverse_sum = sum(inverse_list)
            wage_list = inverse_list / inverse_sum
            wage_list = wage_list.values.tolist()
            group_dict = {
                'A': 0,
                'B': 0,
                'C': 0
            }
            for j in range(len(inverse_list)):
                group_dict[groups[j]] += wage_list[j]
            defined_group = max(group_dict, key=group_dict.get)
            self.test_data.loc[i, 'group'] = defined_group
            if defined_group == cp_test_groups[i]:
                correct += 1
        self.accuracy = np.round(correct / len(cp_test_groups) * 100, 2)
        return None
    
    def plot_knn(self, dist_func):
        self.knn_calculation(dist_func)
        fig, axs = plt.subplots(2, 2, figsize=(10, 8))

        a_group_data = self.data[self.data['group'] == 'A']
        b_group_data = self.data[self.data['group'] == 'B']
        c_group_data = self.data[self.data['group'] == 'C']
